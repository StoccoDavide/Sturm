75 void build(Poly<Real> const & p) {
76 this->m_intervals.clear();
77 Poly<Real> dp, q, r;
78 p.derivative(dp);
79 this->m_sequence.clear();
80 this->m_sequence.reserve(p.order());
81 this->m_sequence.emplace_back(p); this->m_sequence.back().adjust_degree();
82 this->m_sequence.emplace_back(dp); this->m_sequence.back().adjust_degree();
83 Integer n_sequence{1};
84 while (true) {
85 Sturm::divide<Real>(this->m_sequence[n_sequence - 1], this->m_sequence[n_sequence], q, r);
86 if (r.order() <= 0) {break;}
87 this->m_sequence.emplace_back(-r);
88 ++n_sequence;
89 }
90 // Divide by GCD
91 for (Integer i{0}; i < n_sequence; ++i) {
92 Sturm::divide<Real>(this->m_sequence[i], this->m_sequence.back(), q, r);
93 q.normalize();
94 this->m_sequence[i] = q;
95 }
96 this->m_sequence.back().set_scalar(1);
97 }

116 Integer sign_variations(Real x, bool & on_root ) const {
117 Integer sign_var{0};
118 Integer last_sign{0};
119 Integer n_poly{Integer(this->m_sequence.size())};
120 Real v{this->m_sequence[0].evaluate(x)};
121 on_root = false;
122 if (v > 0) {last_sign = 1;}
123 else if (v < 0) {last_sign = -1;}
124 else {
125 on_root = true;
126 last_sign = 0;
127 }
128 for (Integer i{1}; i < n_poly; ++i) {
129 v = this->m_sequence[i].evaluate(x);
130 if (v > 0) {
131 if (last_sign == -1) {++sign_var;}
132 last_sign = 1;
133 } else if (v < 0) {
134 if (last_sign == 1) {++sign_var;}
135 last_sign = -1;
136 }
137 }
138 return sign_var;
139 }

145 Integer separate_roots(Real a_in, Real b_in) {
146 this->m_intervals.clear();
147 this->m_intervals.reserve(this->m_sequence.size());
148
149 Interval I_0, I_1;
150 this->m_a = I_0.a = a_in;
151 this->m_b = I_0.b = b_in;
152
153 I_0.va = this->sign_variations(I_0.a, I_0.a_on_root);
154 I_0.vb = this->sign_variations(I_0.b, I_0.b_on_root);
155
156 Integer n_roots{std::abs(I_0.va - I_0.vb)};
157
158 if (n_roots <= 1) {
159 if (n_roots == 1 && !I_0.a_on_root && !I_0.b_on_root) {
160 this->m_intervals.push_back(I_0);
161 }
162 if (I_0.a_on_root) {
163 I_1.a = I_1.b = I_0.a;
164 I_1.va = I_1.vb = I_0.va;
165 I_1.a_on_root = I_1.b_on_root = true;
166 this->m_intervals.push_back(I_1);
167 }
168 if (I_0.b_on_root) {
169 I_1.a = I_1.b = I_0.b;
170 I_1.va = I_1.vb = I_0.vb;
171 I_1.a_on_root = I_1.b_on_root = true;
172 this->m_intervals.push_back(I_1);
173 }
174 return static_cast<Integer>(m_intervals.size());
175 }
176
177 // Search intervals
178 std::vector<Interval> I_stack; I_stack.clear(); I_stack.reserve(this->m_sequence.size());
179 I_stack.push_back(I_0);
180 while (I_stack.size() > 0) {
181 I_0 = I_stack.back(); I_stack.pop_back();
182 // Check if the interval has a single root
183 n_roots = std::abs(I_0.va - I_0.vb);
184 if (n_roots <= 1) {
185 if (I_0.a_on_root) {
186 I_0.b = I_0.a;
187 I_0.vb = I_0.va;
188 I_0.b_on_root = true;
189 this->m_intervals.push_back(I_0);
190 } else if (I_0.b_on_root) {
191 I_0.a = I_0.b;
192 I_0.va = I_0.vb;
193 I_0.a_on_root = true;
194 this->m_intervals.push_back(I_0);
195 } else if (n_roots == 1) {
196 this->m_intervals.push_back(I_0);
197 }
198 } else if (abs(I_0.b-I_0.a) <= static_cast<Real>(10.0)*EPSILON*std::max(static_cast<Real>(1.0),
199 std::max(abs(I_0.b),abs(I_0.a)))) {
200 I_1.a = I_1.b = I_0.a;
201 I_1.va = I_1.vb = 0;
202 I_1.a_on_root = I_1.b_on_root = true;
203 I_stack.push_back(I_1);
204 } else {
205 Real c{(I_0.a + I_0.b)/static_cast<Real>(2.0)};
206 bool c_on_root;
207 Integer vc{this->sign_variations(c, c_on_root)};
208 // Check the interval [a, c]
209 if (I_0.va != vc || c_on_root || I_0.a_on_root) {
210 if (c < I_1.b) { // Check if it is a true reduction
211 I_1.a = I_0.a; I_1.va = I_0.va; I_1.a_on_root = I_0.a_on_root;
212 I_1.b = c; I_1.vb = vc; I_1.b_on_root = c_on_root;
213 I_stack.push_back(I_1);
214 } else if (c_on_root) {
215 I_1.a = I_1.b = c;
216 I_1.a_on_root = I_1.b_on_root = true;
217 I_1.va = I_1.vb = 0;
218 I_stack.push_back(I_1);
219 } else if (I_0.a_on_root) {
220 I_1.a = I_1.b = I_0.a;
221 I_1.a_on_root = I_1.b_on_root = true;
222 I_1.va = I_1.vb = 0;
223 I_stack.push_back(I_1);
224 }
225 }
226 // Check the interval [c, b]
227 if (I_0.vb != vc || I_0.b_on_root) {
228 if (c > I_0.a) {
229 I_1.a = c; I_1.va = vc; I_1.a_on_root = c_on_root;
230 I_1.b = I_0.b; I_1.vb = I_0.vb; I_1.b_on_root = I_0.b_on_root;
231 I_stack.push_back(I_1);
232 } else if (I_0.b_on_root) {
233 I_1.a = I_1.b = I_0.b;
234 I_1.a_on_root = I_1.b_on_root = true;
235 I_1.va = I_1.vb = 0;
236 I_stack.push_back(I_1);
237 }
238 }
239 }
240 }
241 // Sort intervals
242 std::sort(this->m_intervals.begin(), this->m_intervals.end(),
243 [](Interval const & I_a, Interval const & I_b) {return I_a.a < I_b.a;});
244 return static_cast<Integer>(m_intervals.size());
245 }

253 // Cauchy's bounds for roots
254 Real leading_coeff{this->m_sequence[0].leading_coeff()};
255 Real bound{static_cast<Real>(1.0) + this->m_sequence[0].cwiseAbs().maxCoeff() / std::abs(leading_coeff)};
