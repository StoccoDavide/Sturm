147 std::string to_string() const {
148
149 if (this->order() <= 0) {return "(empty polynomial)";}
150 if (this->order() == 1) {return std::to_string(this->coeff(0));};
151 if ((*this).cwiseAbs().maxCoeff() == 0) {return "0";};
152
153 bool empty{true}; // true means that all the coefficients are zero
154 std::string s{""}; // sign
155 Real c{0}; // coefficient
156 std::string e{""}; // exponent
157 std::string res{""};
158
159 // Check if the first coefficient is different from zero
160 if (this->coeff(0) != 0) {
161 res = std::to_string(this->coeff(0));
162 empty = false;
163 }
164
165 for (Integer i{1}; i < this->order(); ++i) {
166 // If the coefficient is negative...
167 if (this->coeff(i) < 0) {
168 // and if the previous coefficients are null...
169 if (empty) {
170 s = ""; // ...do not write the sign
171 c = this->coeff(i);
172 empty = false;
173 } else {
174 s = " - "; // ...otherwise write the sign as a separator
175 c = -this->coeff(i); // ...and change the sign of the coefficient
176 }
177
178 } else if (this->coeff(i) > 0) {
179 // If the coefficient is positive...
180 c = this->coeff(i);
181 // and if the previous coefficients are null...
182 if (empty) {
183 s = ""; // ...do not write the sign
184 empty = false;
185 } else {
186 s = " + "; // ...otherwise write the sign as a separator
187 }
188
189 } else {
190 // If the coefficient is zero, skip to the next
191 continue;
192 }
193
194 // If the exponent is 1, do not write it
195 if (i == 1) {e = "x";}
196 else {e = "x^" + std::to_string(i);}
197
198 // If the coefficient is 1, do not write it
199 if (c == 1) {res += s; res += e;}
200 else {res += s + std::to_string(c) + e;}
201 }
202 return res;
203 }

642 template <typename Real>
643 inline void divide(Poly<Real> const & p_1, Poly<Real> const & p_2, Poly<Real> & q, Poly<Real> & r) {
644 // Scale polynomials as p_1_norm(x) = p_1(x) / scale_p_1, and p_2_norm(x) = p_2(x) / scale_p_2
645 Poly<Real> p_1_norm(p_1), p_2_norm(p_2);
646 Real scale_p_1{p_1_norm.normalize()};
647 Real scale_p_2{p_2_norm.normalize()};
648
649 // p_1_norm(x) = p_2_norm(x) * q(x) + r(x)
650 r = p_1_norm;
651 Real leading_b_norm{p_2_norm.leading_coeff()};
652 Integer d{r.order() - p_2_norm.order()};
653 if (d < 0) {
654 // p_1_norm(x) = p_2_norm(x) + r(x)
655 q.set_scalar(1);
656 r = p_1_norm - p_2_norm;
657 } else {
658 Integer r_degree{r.degree()};
659 q.set_order(d+1);
660
661 STURM_ASSERT(std::abs(leading_b_norm) > Poly<Real>::EPSILON,
662 "Sturm::Poly::divide(...): leading coefficient of p_2(x) is 0.");
663
664 while (d >= 0 && r_degree >= 0) {
665 Real leading_r = {r(r_degree)};
666 Real ratio{leading_r/leading_b_norm};
667 q.coeffRef(d) = ratio;
668 r.segment(d, p_2_norm.degree()).noalias() -= ratio*p_2_norm.head(p_2_norm.degree());
669 r.coeffRef(r_degree) = 0;
670 --r_degree;
671 --d;
672 }
673
674 // Do not purge remainder: this can be done externally with r.purge(eps);
675 r.adjust_degree();
676 }
677
678 /*
679 Scale back polinomials:
680 - p_1_norm(x) = p_2_norm(x) * q(x) + r(x)
681 - p_1(x) / scale_p_1 = p_2(x) / scale_p_2 * q(x) + r(x)
682 - p_1(x) = p_2(x) * (scale_p_1/scale_p_2) * q(x) + scale_p_1*r(x)
683 */
684 q *= scale_p_1/scale_p_2;
685 r *= scale_p_1;
686 }

696 template <typename Real>
697 inline void GCD(Poly<Real> const & p_1, Poly<Real> const & p_2, Poly<Real> & gcd, Real eps = Poly<Real>::EPSILON)
698 {
699 if (p_2.order() > 0) {
700 Poly<Real> q, r;
701 Sturm::divide(p_1, p_2, q, r);
702 r.purge(eps);
703 Sturm::GCD(p_2, r, gcd, eps);
704 } else {
705 gcd = p_1;
706 }
707 gcd.normalize();
708 }
