141 std::string to_string() const {
142
143 if (this->order() <= 0) {return "(empty polynomial)";}
144 if (this->order() == 1) {return std::to_string(this->coeff(0));};
145 if ((*this).cwiseAbs().maxCoeff() == 0) {return "0";};
146
147 bool empty{true}; // true means that all the coefficients are zero
148 std::string s{""}; // sign
149 Real c{0}; // coefficient
150 std::string e{""}; // exponent
151 std::string res{""};
152
153 // Check if the first coefficient is different from zero
154 if (this->coeff(0) != 0) {
155 res = std::to_string(this->coeff(0));
156 empty = false;
157 }
158
159 for (Integer i{1}; i < this->order(); ++i) {
160 // If the coefficient is negative...
161 if (this->coeff(i) < 0) {
162 // and if the previous coefficients are null...
163 if (empty) {
164 s = ""; // ...do not write the sign
165 c = this->coeff(i);
166 empty = false;
167 } else {
168 s = " - "; // ...otherwise write the sign as a separator
169 c = -this->coeff(i); // ...and change the sign of the coefficient
170 }
171
172 } else if (this->coeff(i) > 0) {
173 // If the coefficient is positive...
174 c = this->coeff(i);
175 // and if the previous coefficients are null...
176 if (empty) {
177 s = ""; // ...do not write the sign
178 empty = false;
179 } else {
180 s = " + "; // ...otherwise write the sign as a separator
181 }
182
183 } else {
184 // If the coefficient is zero, skip to the next
185 continue;
186 }
187
188 // If the exponent is 1, do not write it
189 if (i == 1) {e = "x";}
190 else {e = "x^" + std::to_string(i);}
191
192 // If the coefficient is 1, do not write it
193 if (c == 1) {res += s; res += e;}
194 else {res += s + std::to_string(c) + e;}
195 }
196 return res;
197 }

617 inline void divide(Poly const & p_1, Poly const & p_2, Poly & q, Poly & r) {
618 // Scale polynomials as p_1_norm(x) = p_1(x) / scale_p_1, and p_2_norm(x) = p_2(x) / scale_p_2
619 Poly p_1_norm(p_1), p_2_norm(p_2);
620 Real scale_p_1{p_1_norm.normalize()};
621 Real scale_p_2{p_2_norm.normalize()};
622
623 // p_1_norm(x) = p_2_norm(x) * q(x) + r(x)
624 r = p_1_norm;
625 Real leading_b_norm{p_2_norm.leading_coeff()};
626 Integer d{r.order() - p_2_norm.order()};
627 if (d < 0) {
628 // p_1_norm(x) = p_2_norm(x) + r(x)
629 q.set_scalar(1);
630 r = p_1_norm - p_2_norm;
631 } else {
632 Integer r_degree{r.degree()};
633 q.set_order(d+1);
634
635 STURM_ASSERT(std::abs(leading_b_norm) > EPSILON,
636 "Sturm::Poly::divide(...): leading coefficient of p_2(x) is 0.");
637
638 while (d >= 0 && r_degree >= 0) {
639 Real leading_r = {r(r_degree)};
640 Real ratio{leading_r/leading_b_norm};
641 q.coeffRef(d) = ratio;
642 r.segment(d, p_2_norm.degree()).noalias() -= ratio*p_2_norm.head(p_2_norm.degree());
643 r.coeffRef(r_degree) = 0;
644 --r_degree;
645 --d;
646 }
647
648 // Do not purge remainder: this can be done externally with r.purge(eps);
649 r.adjust_degree();
650 }
651
652 /*
653 Scale back polinomials:
654 - p_1_norm(x) = p_2_norm(x) * q(x) + r(x)
655 - p_1(x) / scale_p_1 = p_2(x) / scale_p_2 * q(x) + r(x)
656 - p_1(x) = p_2(x) * (scale_p_1/scale_p_2) * q(x) + scale_p_1*r(x)
657 */
658 q *= scale_p_1/scale_p_2;
659 r *= scale_p_1;
660 }

669 inline void GCD(Poly const & p_1, Poly const & p_2, Poly & gcd, Real eps = EPSILON)
670 {
671 if (p_2.order() > 0) {
672 Poly q, r;
673 Sturm::divide(p_1, p_2, q, r);
674 r.purge(eps);
675 Sturm::GCD(p_2, r, gcd, eps);
676 } else {
677 gcd = p_1;
678 }
679 gcd.normalize();
680 }
