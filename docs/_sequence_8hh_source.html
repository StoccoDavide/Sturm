88 void build(const Poly<Real> &p) {
89 this->m_intervals.clear();
90 Poly<Real> dp, q, r;
91 p.derivative(dp);
92 this->m_sequence.clear();
93 this->m_sequence.reserve(p.order());
94 this->m_sequence.emplace_back(p);
95 this->m_sequence.back().adjust_degree();
96 this->m_sequence.emplace_back(dp);
97 this->m_sequence.back().adjust_degree();
98 Integer n_sequence{1};
99 while (true) {
100 Sturm::divide<Real>(this->m_sequence[n_sequence - 1],
101 this->m_sequence[n_sequence],
102 q,
103 r);
104 if (r.order() <= 0) {
105 break;
106 }
107 this->m_sequence.emplace_back(-r);
108 ++n_sequence;
109 }
110 // Divide by GCD
111 for (Integer i{0}; i < n_sequence; ++i) {
112 Sturm::divide<Real>(this->m_sequence[i], this->m_sequence.back(), q, r);
113 q.normalize();
114 this->m_sequence[i] = q;
115 }
116 this->m_sequence.back().set_scalar(1);
117 }

139 Integer sign_variations(Real x, bool &on_root) const {
140 Integer sign_var{0};
141 Integer last_sign{0};
142 Integer n_poly{Integer(this->m_sequence.size())};
143 Real v{this->m_sequence[0].evaluate(x)};
144 on_root = false;
145 if (v > 0) {
146 last_sign = 1;
147 } else if (v < 0) {
148 last_sign = -1;
149 } else {
150 on_root = true;
151 last_sign = 0;
152 }
153 for (Integer i{1}; i < n_poly; ++i) {
154 v = this->m_sequence[i].evaluate(x);
155 if (v > 0) {
156 if (last_sign == -1) {
157 ++sign_var;
158 }
159 last_sign = 1;
160 } else if (v < 0) {
161 if (last_sign == 1) {
162 ++sign_var;
163 }
164 last_sign = -1;
165 }
166 }
167 return sign_var;
168 }

177 Integer separate_roots(Real a_in, Real b_in) {
178 this->m_intervals.clear();
179 this->m_intervals.reserve(this->m_sequence.size());
180
181 Interval I_0, I_1;
182 this->m_a = I_0.a = a_in;
183 this->m_b = I_0.b = b_in;
184
185 I_0.va = this->sign_variations(I_0.a, I_0.a_on_root);
186 I_0.vb = this->sign_variations(I_0.b, I_0.b_on_root);
187
188 Integer n_roots{std::abs(I_0.va - I_0.vb)};
189
190 if (n_roots <= 1) {
191 if (n_roots == 1 && !I_0.a_on_root && !I_0.b_on_root) {
192 this->m_intervals.push_back(I_0);
193 }
194 if (I_0.a_on_root) {
195 I_1.a = I_1.b = I_0.a;
196 I_1.va = I_1.vb = I_0.va;
197 I_1.a_on_root = I_1.b_on_root = true;
198 this->m_intervals.push_back(I_1);
199 }
200 if (I_0.b_on_root) {
201 I_1.a = I_1.b = I_0.b;
202 I_1.va = I_1.vb = I_0.vb;
203 I_1.a_on_root = I_1.b_on_root = true;
204 this->m_intervals.push_back(I_1);
205 }
206 return static_cast<Integer>(m_intervals.size());
207 }
208
209 // Search intervals
210 std::vector<Interval> I_stack;
211 I_stack.clear();
212 I_stack.reserve(this->m_sequence.size());
213 I_stack.push_back(I_0);
214 while (I_stack.size() > 0) {
215 I_0 = I_stack.back();
216 I_stack.pop_back();
217 // Check if the interval has a single root
218 n_roots = std::abs(I_0.va - I_0.vb);
219 if (n_roots <= 1) {
220 if (I_0.a_on_root) {
221 I_0.b = I_0.a;
222 I_0.vb = I_0.va;
223 I_0.b_on_root = true;
224 this->m_intervals.push_back(I_0);
225 } else if (I_0.b_on_root) {
226 I_0.a = I_0.b;
227 I_0.va = I_0.vb;
228 I_0.a_on_root = true;
229 this->m_intervals.push_back(I_0);
230 } else if (n_roots == 1) {
231 this->m_intervals.push_back(I_0);
232 }
233 } else if (std::abs(I_0.b - I_0.a) <=
234 static_cast<Real>(10.0) *
235 std::numeric_limits<Real>::epsilon() *
236 std::max(static_cast<Real>(1.0),
237 std::max(std::abs(I_0.b), std::abs(I_0.a)))) {
238 I_1.a = I_1.b = I_0.a;
239 I_1.va = I_1.vb = 0;
240 I_1.a_on_root = I_1.b_on_root = true;
241 I_stack.push_back(I_1);
242 } else {
243 Real c{(I_0.a + I_0.b) / static_cast<Real>(2.0)};
244 bool c_on_root;
245 Integer vc{this->sign_variations(c, c_on_root)};
246 // Check the interval [a, c]
247 if (I_0.va != vc || c_on_root || I_0.a_on_root) {
248 if (c < I_0.b) { // Check if it is a true reduction
249 I_1.a = I_0.a;
250 I_1.va = I_0.va;
251 I_1.a_on_root = I_0.a_on_root;
252 I_1.b = c;
253 I_1.vb = vc;
254 I_1.b_on_root = c_on_root;
255 I_stack.push_back(I_1);
256 } else if (c_on_root) {
257 I_1.a = I_1.b = c;
258 I_1.a_on_root = I_1.b_on_root = true;
259 I_1.va = I_1.vb = 0;
260 I_stack.push_back(I_1);
261 } else if (I_0.a_on_root) {
262 I_1.a = I_1.b = I_0.a;
263 I_1.a_on_root = I_1.b_on_root = true;
264 I_1.va = I_1.vb = 0;
265 I_stack.push_back(I_1);
266 }
267 }
268 // Check the interval [c, b]
269 if (I_0.vb != vc || I_0.b_on_root) {
270 if (c > I_0.a) {
271 I_1.a = c;
272 I_1.va = vc;
273 I_1.a_on_root = c_on_root;
274 I_1.b = I_0.b;
275 I_1.vb = I_0.vb;
276 I_1.b_on_root = I_0.b_on_root;
277 I_stack.push_back(I_1);
278 } else if (I_0.b_on_root) {
279 I_1.a = I_1.b = I_0.b;
280 I_1.a_on_root = I_1.b_on_root = true;
281 I_1.va = I_1.vb = 0;
282 I_stack.push_back(I_1);
283 }
284 }
285 }
286 }
287 // Sort intervals
288 std::sort(this->m_intervals.begin(),
289 this->m_intervals.end(),
290 [](const Interval &I_a, const Interval &I_b) {
291 return I_a.a < I_b.a;
292 });
293 return static_cast<Integer>(m_intervals.size());
294 }

304 Integer separate_roots() {
305 // Cauchy's bounds for roots
306 Real leading_coeff{this->m_sequence[0].leading_coeff()};
307 Real bound{static_cast<Real>(1.0) +
308 this->m_sequence[0].cwiseAbs().maxCoeff() /
309 std::abs(leading_coeff)};
310 return separate_roots(-bound, bound);

337 template <typename SolveFunction>
338 Vector refine_roots(SolveFunction &&solve_function, bool verbose = false) {
339 auto function = std::function<Real(Real x)>([this](Real x) {
340 return this->m_sequence[0].evaluate(x);
341 });
342 Vector roots(this->m_intervals.size());
343 Integer n{0};
344 bool converged{false};
345 for (auto &I : this->m_intervals) {
346 Real &r{roots.coeffRef(n++)};
347 if (I.a_on_root) {
348 r = I.a;
349 } else if (I.b_on_root) {
350 r = I.b;
351 } else {
352 converged = solve_function(I.a, I.b, function, r);
353 STURM_ASSERT_WARNING(
354 converged || !verbose,
355 "Sturm::Sequence::refine_roots(...): failed at interval n = "
356 << n);
357 }
358 }
359 return roots;
360 }

371 template <typename Real>
372 inline std::ostream &operator<<(std::ostream &os, const Sequence<Real> &s) {
373 // Print the Sturm sequence
374 os << "Sturm sequence" << std::endl;
375 for (Integer i{0}; i < s.length(); ++i) {
376 os << "P_" << i << "(x) = " << s.get(i) << std::endl;
377 }
378 // Print the roots
379 Integer n{s.roots_number()};
380 if (n > 0) {
381 os << "roots separation for interval [" << s.a() << "," << s.b() << "]"
382 << std::endl;
383 for (Integer i{0}; i < n; ++i) {
384 const typename Sequence<Real>::Interval &I = s.interval(i);
385 os << "I = [" << I.a << ", " << I.b << "], V = [" << I.va << ", "
386 << I.vb << "]" << std::endl;
387 }
388 }
389 return os;
390 }
